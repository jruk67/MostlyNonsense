SELECT dbo_PERSON.PERSONNUM AS Clock_Num, dbo_WFCTOTAL.EMPLOYEEID, dbo_WFCTOTAL.APPLYDTM, dbo_LABORACCT.LABORLEV4NM, dbo_LABORACCT.LABORLEV4DSC, dbo_LABORACCT.LABORLEV5DSC, dbo_PAYCODE.NAME, dbo_PERSON.LASTNM, dbo_PERSON.FIRSTNM, dbo_PERSON.FULLNM, dbo_VP_EMPLOYEE.HOMELABORLEVELNAME3, Sum(dbo_WFCTOTAL.DURATIONSECSQTY) AS SumOfDURATIONSECSQTY, Sum([DURATIONSECSQTY])/3600 AS Hours
FROM (((dbo_WFCTOTAL INNER JOIN dbo_PAYCODE ON dbo_WFCTOTAL.PAYCODEID = dbo_PAYCODE.PAYCODEID) INNER JOIN dbo_LABORACCT ON dbo_WFCTOTAL.LABORACCTID = dbo_LABORACCT.LABORACCTID) INNER JOIN dbo_PERSON ON dbo_WFCTOTAL.EMPLOYEEID = dbo_PERSON.PERSONID) INNER JOIN dbo_VP_EMPLOYEE ON dbo_PERSON.PERSONID = dbo_VP_EMPLOYEE.PERSONID
WHERE (((dbo_WFCTOTAL.APPLYDTM)>=? And (dbo_WFCTOTAL.APPLYDTM)<=?) AND ((dbo_VP_EMPLOYEE.HOMELABORLEVELNAME3) Like ?))
GROUP BY dbo_PERSON.PERSONNUM, dbo_WFCTOTAL.EMPLOYEEID, dbo_WFCTOTAL.APPLYDTM, dbo_LABORACCT.LABORLEV4NM, dbo_LABORACCT.LABORLEV4DSC, dbo_LABORACCT.LABORLEV5DSC, dbo_PAYCODE.NAME, dbo_PERSON.LASTNM, dbo_PERSON.FIRSTNM, dbo_PERSON.FULLNM, dbo_VP_EMPLOYEE.HOMELABORLEVELNAME3;
